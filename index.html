Featured snippet: animated metrics card

Plug this component into your dashboard and customize colors, labels, and values instantly.

const metrics = [
  { label: "Signups", value: 1920, trend: "+12%" },
  { label: "Conversion", value: "8.4%", trend: "+1.1%" },
  { label: "Revenue", value: "$12.4k", trend: "+18%" }
];

function renderMetrics(items) {
  return items.map((item) => `
    <article class="metric-card">
      <h4>${item.label}</h4>
      <strong>${item.value}</strong>
      <span>${item.trend} this week</span>
    </article>`).join("");
}
